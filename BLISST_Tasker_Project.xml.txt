  <script>
    const titleInput = document.getElementById('list-title');
    const container = document.getElementById('items-container');
    const syncBtn = document.getElementById('sync-btn');
    const addItemBtn = document.getElementById('add-item-btn');
    const status = document.getElementById('status');
    const connectionBadge = document.getElementById('connection-badge');
    const connectionText = document.getElementById('connection-text');
    let draggedRow = null;

    // --- UPDATED: Live Connection Polling ---
    async function pollSessionStatus() {
      try {
        const res = await fetch('/api/health');
        const data = await res.json();
        
        // Now checks specifically for the glasses flag
        if (data.glassesConnected === true) {
          connectionBadge.classList.add('connected');
          connectionText.textContent = 'Glasses Active & Listening';
        } else {
          connectionBadge.classList.remove('connected');
          connectionText.textContent = 'Glasses disconnected';
        }
      } catch (e) {
        connectionBadge.classList.remove('connected');
        connectionText.textContent = 'Disconnected from server';
      }
    }
    pollSessionStatus();
    setInterval(pollSessionStatus, 2000);
    // ------------------------------------

    function showStatus(msg, isErr = false) {
      status.textContent = msg;
      status.style.display = 'block';
      status.style.background = isErr ? '#fee2e2' : '#e0f2fe';
      status.style.color = isErr ? '#991b1b' : '#0c4a6e';
      setTimeout(() => status.style.display = 'none', 4000);
    }

    function updateNumbers() {
      container.querySelectorAll('.step-number').forEach((s, i) => {
        s.textContent = String(i + 1).padStart(2, '0');
      });
    }

    function createRow(val = '') {
      const row = document.createElement('div');
      row.className = 'instruction-row';
      row.draggable = true;
      row.innerHTML = `
        <div class="drag-handle">⋮⋮</div>
        <div class="step-number">00</div>
        <input type="text" class="input-field" placeholder="Item..." value="${val}" autocomplete="off">
        <button class="remove-btn">×</button>
      `;

      const input = row.querySelector('input');
      
      input.onkeydown = (e) => {
        if (e.key === 'Enter') {
          e.preventDefault();
          const newRow = createRow();
          row.after(newRow);
          updateNumbers();
          newRow.querySelector('input').focus();
        }
        if (e.key === 'Backspace' && input.value === '' && container.children.length > 1) {
          e.preventDefault();
          const prev = row.previousElementSibling;
          row.remove();
          updateNumbers();
          if (prev) prev.querySelector('input').focus();
        }
      };

      row.querySelector('.remove-btn').onclick = () => {
        if (container.children.length > 1) { row.remove(); updateNumbers(); }
        else { input.value = ''; input.focus(); }
      };

      row.ondragstart = () => { draggedRow = row; setTimeout(() => row.classList.add('dragging'), 0); };
      row.ondragend = () => { row.classList.remove('dragging'); updateNumbers(); };
      row.ondragover = (e) => {
        e.preventDefault();
        const rect = row.getBoundingClientRect();
        const mid = rect.y + rect.height / 2;
        row.classList.remove('drag-over-top', 'drag-over-bottom');
        row.classList.add(e.clientY > mid ? 'drag-over-bottom' : 'drag-over-top');
      };
      row.ondragleave = () => row.classList.remove('drag-over-top', 'drag-over-bottom');
      row.ondrop = (e) => {
        row.classList.remove('drag-over-top', 'drag-over-bottom');
        if (row !== draggedRow) {
          const rect = row.getBoundingClientRect();
          if (e.clientY > rect.y + rect.height / 2) row.after(draggedRow);
          else row.before(draggedRow);
        }
      };

      return row;
    }

    titleInput.onkeydown = (e) => {
      if (e.key === 'Enter') {
        e.preventDefault();
        container.querySelector('input').focus();
      }
    };

    addItemBtn.onclick = () => {
      const newRow = createRow();
      container.appendChild(newRow);
      updateNumbers();
      newRow.querySelector('input').focus();
    };

    syncBtn.onclick = async () => {
      const title = titleInput.value.trim();
      const items = Array.from(container.querySelectorAll('.input-field'))
        .map(i => i.value.trim())
        .filter(Boolean);

      if (!title || !items.length) return showStatus("Need a title and items!", true);

      try {
        const res = await fetch('/api/active-list', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            title,
            uid: Math.random().toString(36).slice(2),
            instructions: items.map((text, i) => ({ step: i + 1, instruction: text }))
          })
        });
        if (res.ok) showStatus("Synced to glasses!");
        else throw new Error();
      } catch (e) { showStatus("Sync failed", true); }
    };

    async function init() {
      try {
        const res = await fetch('/api/active-list');
        if (res.ok) {
          const data = await res.json();
          titleInput.value = data.title || '';
          data.instructions.forEach(i => container.appendChild(createRow(i.instruction)));
        } else {
          container.appendChild(createRow());
        }
      } catch (e) {
        container.appendChild(createRow());
      }
      updateNumbers();
    }

    init();
  </script>
